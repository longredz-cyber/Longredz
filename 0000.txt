local function SimpleHash(str)
	local hash = 0
	for i = 1, #str do
		local char = string.byte(str, i)
		hash = (hash * 31 + char) % 65536
	end
	return string.format("%04X", hash)
end

function GenerateChecksum(part1, part2, part3, duration)
	local data = part1 .. "|" .. part2 .. "|" .. part3 .. "|" .. duration .. "|" .. SECRET_SALT
	return SimpleHash(data)
end

local function IsGeneratedKeyValid(key)
	-- Kiểm tra blacklist trước
	if IsKeyBlacklisted(key) then return false end
	
	local parts = {}
	for part in string.gmatch(key, "[^-]+") do
		table.insert(parts, part)
	end
	
	if #parts ~= 5 then return false end
	
	local c1 = parts[1]
	local c2 = parts[2]
	local c3 = parts[3]
	local durationCode = parts[4]
	local checksum = parts[5]
	
	local durationValue = nil
	if durationCode == "24H" then durationValue = "24hours"
	elseif durationCode == "7D" then durationValue = "7days"
	elseif durationCode == "30D" then durationValue = "30days"
	elseif durationCode == "FVR" then durationValue = "forever"
	else return false end
	
	local expectedChecksum = GenerateChecksum(c1, c2, c3, durationValue)
	if checksum ~= expectedChecksum then return false end
	
	if durationValue == "forever" then return true
	
	elseif durationValue == "24hours" then
		-- Kiểm tra trong file lưu trữ
		local filename = "LongRedz_Keys/Generated/" .. key .. ".json"
		if isfile(filename) then
			local success, content = pcall(function() return readfile(filename) end)
			if success and content then
				local data = HttpService:JSONDecode(content)
				return (os.time() - data.CreatedAt) < 86400
			end
		end
		return (os.time() % 86400) < 86400
		
	elseif durationValue == "7days" then
		local filename = "LongRedz_Keys/Generated/" .. key .. ".json"
		if isfile(filename) then
			local success, content = pcall(function() return readfile(filename) end)
			if success and content then
				local data = HttpService:JSONDecode(content)
				return (os.time() - data.CreatedAt) < 604800
			end
		end
		return true
		
	elseif durationValue == "30days" then
		local filename = "LongRedz_Keys/Generated/" .. key .. ".json"
		if isfile(filename) then
			local success, content = pcall(function() return readfile(filename) end)
			if success and content then
				local data = HttpService:JSONDecode(content)
				return (os.time() - data.CreatedAt) < 2592000
			end
		end
		return true
	end
	
	return false
end

local function GetDailyKey()
	local daySeed = math.floor(os.time() / 86400)
	math.randomseed(daySeed)
	local index = math.random(1, #validKeys)
	local dailyKey = validKeys[index]
	math.randomseed(os.time())
	return dailyKey
end

local function IsValidKey(inputKey)
	if inputKey == adminKey then return true end
	
	local currentKey = GetDailyKey()
	if inputKey == currentKey then return true end
	
	local yesterdaySeed = math.floor(os.time() / 86400) - 1
	math.randomseed(yesterdaySeed)
	local yesterdayIndex = math.random(1, #validKeys)
	local yesterdayKey = validKeys[yesterdayIndex]
	math.randomseed(os.time())
	
	if inputKey == yesterdayKey then
		local secondsIntoDay = os.time() % 86400
		if secondsIntoDay < 3600 then return true end
	end
	
	if IsGeneratedKeyValid(inputKey) then return true end
	
	return false
end

	local dragging2, dragInput2, dragStart2, startPos2
	local function update(input)
		local delta = input.Position - dragStart2
		BaseHolder.Position = UDim2.new(startPos2.X.Scale, startPos2.X.Offset + delta.X, startPos2.Y.Scale, startPos2.Y.Offset + delta.Y)
	end
	BaseHolder.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging2 = true
			dragStart2 = input.Position
			startPos2 = BaseHolder.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then dragging2 = false end
			end)
		end
	end)
	BaseHolder.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput2 = input
		end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput2 and dragging2 then update(input) end
	end)

		local closeTween = TweenService:Create(BaseHolder, TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.In), {Size = UDim2.new(0, 0, 0, 0)})
		closeTween:Play()
		closeTween.Completed:Connect(function() ManagerMenu:Destroy() end)
	end)

	local function generateKey()
		local c1 = C1_Box.Text ~= "" and C1_Box.Text or "DEFAULT"
		local c2 = C2_Box.Text ~= "" and C2_Box.Text or "KEY"
		local c3 = C3_Box.Text ~= "" and C3_Box.Text or "GEN"
		local sep = SepBox.Text ~= "" and SepBox.Text or "-"
		
		local durationValue = selectedDuration.Value
		
		local durationCode = ""
		local durationName = ""
		if durationValue == "24hours" then durationCode = "24H" durationName = "24 Giờ"
		elseif durationValue == "7days" then durationCode = "7D" durationName = "7 Ngày"
		elseif durationValue == "30days" then durationCode = "30D" durationName = "30 Ngày"
		elseif durationValue == "forever" then durationCode = "FVR" durationName = "Vĩnh viễn"
		end
		
		local checksum = GenerateChecksum(c1, c2, c3, durationValue)
		local key = c1 .. sep .. c2 .. sep .. c3 .. sep .. durationCode .. sep .. checksum
		
		return key, durationName
	end

	local function StartKeyWait()
		if isWaiting then return end
		
		local rand = math.random(1, 100)
		local waitTime
		if rand <= 25 then waitTime = 60
		elseif rand <= 60 then waitTime = 120
		else waitTime = 300 end
		
		isWaiting = true
		waitEndTime = os.time() + waitTime
		
		GetKeyNowButton.Visible = false
		KeyFrame.Visible = false
		CopyButton.Visible = false
		StatusFrame.Visible = true
		CancelButton.Visible = true
		
		local waitMinutes = math.floor(waitTime / 60)
		local waitSeconds = waitTime % 60
		StatusText.Text = "⏳ ĐANG CHỜ " .. waitMinutes .. " PHÚT " .. (waitSeconds > 0 and waitSeconds .. " GIÂY" or "")
		
		game.StarterGui:SetCore("SendNotification", {
			Title = "Đang chờ ⏳";
			Text = "Vui lòng đợi " .. waitMinutes .. " phút để nhận key!";
			Duration = 5;
		})
		
		coroutine.wrap(function()
			while isWaiting and ContentFrame.Parent do
				local remaining = waitEndTime - os.time()
				if remaining <= 0 then
					isWaiting = false
					StatusFrame.Visible = false
					CancelButton.Visible = false
					KeyFrame.Visible = true
					CopyButton.Visible = true
					GetKeyNowButton.Visible = true
					GetKeyNowButton.Text = "🔄 LẤY KEY KHÁC"
					
					KeyText.Text = currentKey
					local secLeft = 86400 - (os.time() % 86400)
					local hLeft = math.floor(secLeft / 3600)
					local mLeft = math.floor((secLeft % 3600) / 60)
					TimeLeftText.Text = "⏳ Còn: " .. hLeft .. "h " .. mLeft .. "m trước khi đổi key"
					
					game.StarterGui:SetCore("SendNotification", {
						Title = "Đã có key! 🎉";
						Text = "Key: " .. currentKey;
						Duration = 5;
					})
					break
				end
				
				local minLeft = math.floor(remaining / 60)
				local secLeft = remaining % 60
				TimerText.Text = "Còn lại: " .. minLeft .. ":" .. string.format("%02d", secLeft)
				task.wait(1)
			end
		end)()
	end

	local function CancelWait()
		if not isWaiting then return end
		isWaiting = false
		StatusFrame.Visible = false
		CancelButton.Visible = false
		GetKeyNowButton.Visible = true
		GetKeyNowButton.Text = "🎁 LẤY KEY NGAY"
		
		game.StarterGui:SetCore("SendNotification", {
			Title = "Đã hủy ❌";
			Text = "Bạn có thể thử lại!";
			Duration = 3;
		})
	end

	RunService.RenderStepped:Connect(function(deltaTime)
		if not ContentFrame.Parent then return end
		local currentTime = tick()
		RainbowInlay.UIGradient.Rotation = (RainbowInlay.UIGradient.Rotation + (110 * deltaTime)) % 360
		local pulse = math.sin(currentTime * 3)
		TitleGlow.Thickness = (pulse + 1) * 1.2
		for _, star in ipairs(galaxyStars) do
			local brightness = 0.3 + (math.sin(currentTime * star.Speed + star.Phase) + 1) * 0.35
			star.Object.BackgroundTransparency = 1 - brightness
		end
		for _, particle in ipairs(dustParticles2) do
			local dustObj = particle.Object
			if dustObj and dustObj.Parent then
				local currentPos = dustObj.Position
				local newY = currentPos.Y.Scale + (particle.Speed * deltaTime)
				local waveOffset = math.sin(currentTime * 1.5 + particle.Seed) * 0.035
				local newX = particle.BaseX + waveOffset
				if newY > 1 then
					particle.BaseX = math.random(5, 95) / 100
					dustObj.Position = UDim2.new(particle.BaseX, 0, -0.05, 0)
				else
					dustObj.Position = UDim2.new(newX, currentPos.X.Offset, newY, currentPos.Y.Offset)
				end
			end
		end
	end)

	local dragging2, dragInput2, dragStart2, startPos2
	local function update(input)
		local delta = input.Position - dragStart2
		BaseHolder.Position = UDim2.new(startPos2.X.Scale, startPos2.X.Offset + delta.X, startPos2.Y.Scale, startPos2.Y.Offset + delta.Y)
	end
	BaseHolder.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging2 = true
			dragStart2 = input.Position
			startPos2 = BaseHolder.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then dragging2 = false end
			end)
		end
	end)
	BaseHolder.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput2 = input
		end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput2 and dragging2 then update(input) end
	end)

		closeTween:Play()
		closeTween.Completed:Connect(function() CoreMenu:Destroy() end)
	end)
