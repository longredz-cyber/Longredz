local function GetDailyKey()
	local daySeed = math.floor(os.time() / 86400)
	math.randomseed(daySeed)
	local index = math.random(1, #validKeys)
	local dailyKey = validKeys[index]
	math.randomseed(os.time())
	return dailyKey
end

-- Hàm kiểm tra key hợp lệ
local function IsValidKey(inputKey)
	local currentKey = GetDailyKey()
	if inputKey == currentKey then return true end
	
	local yesterdaySeed = math.floor(os.time() / 86400) - 1
	math.randomseed(yesterdaySeed)
	local yesterdayIndex = math.random(1, #validKeys)
	local yesterdayKey = validKeys[yesterdayIndex]
	math.randomseed(os.time())
	
	if inputKey == yesterdayKey then
		local secondsIntoDay = os.time() % 86400
		if secondsIntoDay < 3600 then return true end
	end
	return false
end

	local function StartKeyWait()
		if isWaiting then return end
		
		local rand = math.random(1, 100)
		local waitTime
		if rand <= 25 then waitTime = 60
		elseif rand <= 60 then waitTime = 120
		else waitTime = 300 end
		
		isWaiting = true
		waitEndTime = os.time() + waitTime
		
		GetKeyNowButton.Visible = false
		KeyFrame.Visible = false
		CopyButton.Visible = false
		StatusFrame.Visible = true
		CancelButton.Visible = true
		
		local waitMinutes = math.floor(waitTime / 60)
		local waitSeconds = waitTime % 60
		StatusText.Text = "⏳ ĐANG CHỜ " .. waitMinutes .. " PHÚT " .. (waitSeconds > 0 and waitSeconds .. " GIÂY" or "")
		
		game.StarterGui:SetCore("SendNotification", {
			Title = "Đang chờ ⏳";
			Text = "Vui lòng đợi " .. waitMinutes .. " phút để nhận key!";
			Duration = 5;
		})
		
		coroutine.wrap(function()
			while isWaiting and ContentFrame.Parent do
				local remaining = waitEndTime - os.time()
				if remaining <= 0 then
					isWaiting = false
					StatusFrame.Visible = false
					CancelButton.Visible = false
					KeyFrame.Visible = true
					CopyButton.Visible = true
					GetKeyNowButton.Visible = true
					GetKeyNowButton.Text = "🔄 LẤY KEY KHÁC"
					
					KeyText.Text = currentKey
					local secLeft = 86400 - (os.time() % 86400)
					local hLeft = math.floor(secLeft / 3600)
					local mLeft = math.floor((secLeft % 3600) / 60)
					TimeLeftText.Text = "⏳ Còn: " .. hLeft .. "h " .. mLeft .. "m trước khi đổi key"
					
					game.StarterGui:SetCore("SendNotification", {
						Title = "Đã có key! 🎉";
						Text = "Key: " .. currentKey;
						Duration = 5;
					})
					break
				end
				
				local minLeft = math.floor(remaining / 60)
				local secLeft = remaining % 60
				TimerText.Text = "Còn lại: " .. minLeft .. ":" .. string.format("%02d", secLeft)
				task.wait(1)
			end
		end)()
	end

	end)

	TweenService:Create(BaseHolder, TweenInfo.new(0.6, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Size = FinalSize}):Play()

	local dragging2, dragInput2, dragStart2, startPos2
	local function update(input)
		local delta = input.Position - dragStart2
		BaseHolder.Position = UDim2.new(startPos2.X.Scale, startPos2.X.Offset + delta.X, startPos2.Y.Scale, startPos2.Y.Offset + delta.Y)
	end
	BaseHolder.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging2 = true
			dragStart2 = input.Position
			startPos2 = BaseHolder.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then dragging2 = false end
			end)
		end
	end)
	BaseHolder.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput2 = input
		end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput2 and dragging2 then update(input) end
	end)

	CloseMenuButton.MouseButton1Click:Connect(function()
		local closeTween = TweenService:Create(BaseHolder, TweenInfo.new(0.35, Enum.EasingStyle.Back, Enum.EasingDirection.In), {Size = UDim2.new(0, 0, 0, 0)})
		closeTween:Play()
		closeTween.Completed:Connect(function() CoreMenu:Destroy() end)
	end)
end

	end

	isProcessing = false
end)
